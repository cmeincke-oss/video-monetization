Pre-Roll Ad

Google IMA SDK

A skippable pre-roll fires before video content plays. The Google IMA SDK manages ad playback entirely — the React player is not involved. On mobile, this runs inside the WebView with no native SDK required.

Web / React
iOS
Android
// React — IMA SDK pre-roll
// SDK loaded via <script> in <head>

function initIMA() {
  const container = new google.ima
    .AdDisplayContainer(imaDiv, videoEl);

  const loader = new google.ima.AdsLoader(container);
  loader.addEventListener('adsManagerLoaded', onLoaded);
  loader.addEventListener('adError', onError);

  container.initialize(); // must be in user gesture

  const req = new google.ima.AdsRequest();
  req.adTagUrl = AD_TAG_URL;
  req.linearAdSlotWidth  = 480;
  req.linearAdSlotHeight = 270;
  loader.requestAds(req);
}

function onLoaded(e) {
  const mgr = e.getAdsManager(videoEl);
  mgr.addEventListener('contentPauseRequested',
    () => videoEl.pause());
  mgr.addEventListener('contentResumeRequested',
    () => videoEl.play());
  mgr.addEventListener('allAdsCompleted',
    () => videoEl.play());
  mgr.init(480, 270, google.ima.ViewMode.NORMAL);
  mgr.start();
}
